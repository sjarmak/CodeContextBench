type PolicyVerdict string

	// VerdictAllow indicates traffic is allowed.
	VerdictAllow PolicyVerdict = "allow"
	// VerdictDeny indicates traffic is denied.
	VerdictDeny PolicyVerdict = "deny"
)

// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes. It integrates with Cilium's existing
// logging framework and supports JSON-structured output.
type PolicyAuditLogger struct {
	lock.Mutex

	// level is the logrus log level for audit messages.
	level logrus.Level
	// output is the writer for audit log messages; if nil the package-level log is used.
	output io.Writer
	// logger is an optional dedicated logrus.Logger for audit output.
	logger *logrus.Logger
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger at the given log level.
// If output is non-nil, a dedicated JSON-formatted logger writing to output is used.
func NewPolicyAuditLogger(level logrus.Level, output io.Writer) *PolicyAuditLogger {
	pal := &PolicyAuditLogger{
		level:  level,
		output: output,
	}
	if output != nil {
		l := logrus.New()
		l.SetOutput(output)
		l.SetFormatter(&logrus.JSONFormatter{})
		l.SetLevel(level)
		pal.logger = l
	}
	return pal
}
