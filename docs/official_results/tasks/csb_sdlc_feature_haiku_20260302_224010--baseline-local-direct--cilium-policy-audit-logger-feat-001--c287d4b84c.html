type PolicyAuditLogger struct {
	mu       lock.Mutex
	logLevel logrus.Level
	logger   logrus.FieldLogger
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the given log level
// and logger instance.
func NewPolicyAuditLogger(logger logrus.FieldLogger, level logrus.Level) *PolicyAuditLogger {
	if logger == nil {
		logger = log
	}
	return &PolicyAuditLogger{
		logLevel: level,
		logger:   logger,
	}
}

// LogDecision logs a policy decision with identity, policy, verdict, and reason.
// This method is thread-safe.
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict api.Decision, reason string) {
	pal.mu.Lock()
	defer pal.mu.Unlock()

	if !pal.shouldLog() {
		return
	}

	fields := logrus.Fields{
		logfields.Identity:     identity,
		"policyName":           policy,
		"verdict":              verdict.String(),
		"reason":               reason,
	}

	pal.logWithFields(fields, "Policy decision logged")
}
