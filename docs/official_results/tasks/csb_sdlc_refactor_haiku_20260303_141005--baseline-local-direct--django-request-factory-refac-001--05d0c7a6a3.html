import sys

def process_file(filepath):
    """Process a single Python file to rename RequestFactory to TestRequestBuilder"""
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()
    except:
        return False
    
    original = content
    
    # Update imports - handle various import styles
    # from django.test import RequestFactory
    content = re.sub(
        r'from django\.test import ([^\n]*)\bRequestFactory\b',
        lambda m: f'from django.test import {m.group(1)}TestRequestBuilder',
        content
    )
    
    # from django.test.client import RequestFactory
    content = re.sub(
        r'from django\.test\.client import ([^\n]*)\bRequestFactory\b',
        lambda m: f'from django.test.client import {m.group(1)}TestRequestBuilder',
        content
    )
    
    # Remove trailing commas if they exist
    content = re.sub(
        r'(\bTestRequestBuilder),(\s*[\n\)])',
        r'\1\2',
        content
    )
    
    # Update type annotations: RequestFactory -> TestRequestBuilder
    content = re.sub(r'\bRequestFactory\b(?!.*=)', r'TestRequestBuilder', content)
    
    if content != original:
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(content)
        return True
    return False
