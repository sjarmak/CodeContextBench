import (
	"sort"

// DurationTracker records and provides statistics about evaluation durations.
// It is thread-safe for concurrent use.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// Record adds a duration to the tracker.
func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Stats returns the count, average duration, and P99 latency.
func (dt *DurationTracker) Stats() (count int, avg time.Duration, p99 time.Duration) {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	count = len(dt.durations)
	if count == 0 {
		return 0, 0, 0
	}

	// Calculate sum for average
	var sum time.Duration
	for _, d := range dt.durations {
		sum += d
	}
	avg = sum / time.Duration(count)

	// Calculate P99 by sorting
	sorted := make([]time.Duration, count)
	copy(sorted, dt.durations)
	sort.Slice(sorted, func(i, j int) bool {
		return sorted[i] < sorted[j]
	})

	// P99 is the 99th percentile
	p99Index := (99 * count) / 100
	if p99Index >= count {
		p99Index = count - 1
	}
	p99 = sorted[p99Index]

	return count, avg, p99
}

type DurationTracker struct {

func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

func (dt *DurationTracker) Stats() (count int, avg time.Duration, p99 time.Duration) {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	count = len(dt.durations)
	if count == 0 {
		return 0, 0, 0
	}

	var sum time.Duration
	for _, d := range dt.durations {
		sum += d
	}
	avg = sum / time.Duration(count)

	sorted := make([]time.Duration, count)
	copy(sorted, dt.durations)
	
	// Sort manually (simplified)
	for i := 0; i < count; i++ {
		for j := i + 1; j < count; j++ {
			if sorted[j] < sorted[i] {
				sorted[i], sorted[j] = sorted[j], sorted[i]
			}
		}
	}

	p99Index := (99 * count) / 100
	if p99Index >= count {
		p99Index = count - 1
	}
	p99 = sorted[p99Index]

	return count, avg, p99
}
