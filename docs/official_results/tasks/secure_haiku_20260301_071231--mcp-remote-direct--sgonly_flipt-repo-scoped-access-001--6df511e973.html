// Tracker tracks in-process metrics for the evaluation engine.
type Tracker struct {
	mu          sync.RWMutex
	total       int64
	matchCount  int64
	errorCount  int64
}

// NewTracker creates a new evaluation metrics tracker.
func NewTracker() *Tracker {
	return &Tracker{}
}

// RecordEvaluation records a successful evaluation.
func (t *Tracker) RecordEvaluation() {
	t.mu.Lock()
	defer t.mu.Unlock()
	t.total++
}

// RecordMatch records an evaluation that resulted in a match.
func (t *Tracker) RecordMatch() {
	t.mu.Lock()
	defer t.mu.Unlock()
	t.matchCount++
}

// RecordError records an evaluation that resulted in an error.
func (t *Tracker) RecordError() {
	t.mu.Lock()
	defer t.mu.Unlock()
	t.errorCount++
}

// Total returns the total number of evaluations processed.
func (t *Tracker) Total() int64 {
	t.mu.RLock()
	defer t.mu.RUnlock()
	return t.total
}

// MatchCount returns the number of evaluations that resulted in a match.
func (t *Tracker) MatchCount() int64 {
	t.mu.RLock()
	defer t.mu.RUnlock()
	return t.matchCount
}

// ErrorCount returns the number of evaluations that resulted in errors.
func (t *Tracker) ErrorCount() int64 {
	t.mu.RLock()
	defer t.mu.RUnlock()
	return t.errorCount
}

// Metrics returns a snapshot of all current metrics.
func (t *Tracker) Metrics() (total, matchCount, errorCount int64) {
	t.mu.RLock()
	defer t.mu.RUnlock()
	return t.total, t.matchCount, t.errorCount
}
