// PolicyQuotaController manages per-namespace quotas for CiliumNetworkPolicy resources.
type PolicyQuotaController struct {
	log                logrus.FieldLogger
	namespacePolicies  map[string]int
	mu                 sync.Mutex
	maxPerNamespace    int
	cnpResource        resource.Resource[*cilium_v2.CiliumNetworkPolicy]
}

// NewPolicyQuotaController creates a new PolicyQuotaController.
func NewPolicyQuotaController(
	log logrus.FieldLogger,
	cnpResource resource.Resource[*cilium_v2.CiliumNetworkPolicy],
	maxPerNamespace int,
) *PolicyQuotaController {
	return &PolicyQuotaController{
		log:               log,
		namespacePolicies: make(map[string]int),
		maxPerNamespace:   maxPerNamespace,
		cnpResource:       cnpResource,
	}
}

// Run starts the controller's event loop.
func (pqc *PolicyQuotaController) Run(ctx context.Context) error {
	pqc.log.Info("Starting PolicyQuotaController")
	defer pqc.log.Info("Stopped PolicyQuotaController")

	cnpEvents := pqc.cnpResource.Events(ctx)

	for {
		select {
		case <-ctx.Done():
			return ctx.Err()
		case event, ok := <-cnpEvents:
			if !ok {
				return nil
			}

			if event.Kind == resource.Sync {
				pqc.log.Debug("PolicyQuotaController synced")
				event.Done(nil)
				continue
			}

			var err error
			switch event.Kind {
			case resource.Upsert:
				err = pqc.onPolicyUpsert(event.Object)
			case resource.Delete:
				err = pqc.onPolicyDelete(event.Object)
			}
			event.Done(err)
		}
	}
}
