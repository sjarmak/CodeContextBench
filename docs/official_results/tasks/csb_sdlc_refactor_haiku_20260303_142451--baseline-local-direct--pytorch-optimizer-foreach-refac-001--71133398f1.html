def foreach_group_by_device_and_dtype(
    tensor_lists: list,
    with_indices: bool = False,
):
    r"""Group tensors by device and dtype.

    Wrapper around Optimizer._group_tensors_by_device_and_dtype that provides
    a cleaner interface for foreach optimizer implementations.

    Args:
        tensor_lists: List of tensor lists to group
        with_indices: If True, return indices for mapping back to original lists

    Returns:
        grouped_tensors: Dict mapping (device, dtype) to grouped tensor lists
    """
    return Optimizer._group_tensors_by_device_and_dtype(
        tensor_lists, with_indices=with_indices
    )


def get_device_dict(
    tensor: Optional[Tensor] = None,
    device: Optional[torch.device] = None,
) -> Optional[DeviceDict]:
    r"""Create a device dictionary for caching device-specific tensors.

    This is a common pattern in foreach optimizers where scalar parameters
    (like lr, betas, weight_decay) need to be moved to different devices.

    Args:
        tensor: Optional tensor to initialize the dict with
        device: Optional device to initialize the dict with

    Returns:
        DeviceDict or None if tensor is None and device is None
    """
    if tensor is not None:
        return {tensor.device: tensor}
    elif device is not None:
        return {device: None}
    return None
