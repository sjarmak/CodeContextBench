// DurationTracker records and aggregates evaluation durations.
// It is safe for concurrent use.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// NewDurationTracker creates a new DurationTracker.
func NewDurationTracker() *DurationTracker {
	return &DurationTracker{
		durations: make([]time.Duration, 0),
	}
}

// RecordDuration records a single evaluation duration.
func (dt *DurationTracker) RecordDuration(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Stats contains aggregate duration statistics.
type Stats struct {
	Count   int64
	Average time.Duration
	P99     time.Duration
}

// GetStats returns aggregate duration statistics.
func (dt *DurationTracker) GetStats() Stats {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	count := int64(len(dt.durations))
	if count == 0 {
		return Stats{
			Count:   0,
			Average: 0,
			P99:     0,
		}
	}

	// Calculate average
	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}
	average := total / time.Duration(count)

	// Calculate P99
	p99 := dt.calculatePercentile(99)

	return Stats{
		Count:   count,
		Average: average,
		P99:     p99,
	}
}
