**From:** Engineering Manager
**Priority:** P2
**Context:** SRE team needs visibility into evaluation throughput for capacity planning

5. All changes within `internal/server/evaluation/`

// Metrics tracks in-process evaluation metrics.
type Metrics struct {
	mu         sync.RWMutex
	total      int64
	matches    int64
	errors     int64
}

// NewMetrics creates a new Metrics instance.
func NewMetrics() *Metrics {
	return &Metrics{}
}

// RecordEvaluation increments the total evaluation count.
func (m *Metrics) RecordEvaluation() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.total++
}

// RecordMatch increments the match count.
func (m *Metrics) RecordMatch() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.matches++
}

// RecordError increments the error count.
func (m *Metrics) RecordError() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.errors++
}

// GetMetrics returns a snapshot of current metrics.
func (m *Metrics) GetMetrics() (total, matches, errors int64) {
	m.mu.RLock()
	defer m.mu.RUnlock()
	return m.total, m.matches, m.errors
}

// Reset resets all metrics to zero.
func (m *Metrics) Reset() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.total = 0
	m.matches = 0
	m.errors = 0
}
