			name:          "empty original data",

			encryptedData: []byte("encrypted-empty"),
			authenticData: []byte("auth"),
			expectError:   false,
		},
		{
			name:          "large data",
			originalData:  bytes.Repeat([]byte("x"), 10000),
			encryptedData: bytes.Repeat([]byte("y"), 10000),
			authenticData: []byte("large"),
			expectError:   false,
		},
	}

			transformer := &mockTransformer{
				encryptedData: tt.encryptedData,
				decryptedData: tt.originalData,
			}

			ctx := context.Background()
			authCtx := DefaultContext(tt.authenticData)

			// Test encryption
			encrypted, err := transformer.TransformToStorage(ctx, tt.originalData, authCtx)

			}
			if !bytes.Equal(decrypted, tt.originalData) {
				t.Errorf("TransformFromStorage returned %q, want %q", decrypted, tt.originalData)
			}
		})
	}
}
