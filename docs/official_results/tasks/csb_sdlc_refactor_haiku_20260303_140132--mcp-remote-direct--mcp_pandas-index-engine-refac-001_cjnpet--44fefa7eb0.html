#!/usr/bin/env python3
"""
Rename _engine to _index_engine in Index-related files.
Be careful not to rename:
- _engine_type or _engine_types
- _engine_target  
- Other engine references in I/O modules (Excel writers, CSV parsers, etc.)
"""

import re

def rename_engine_property(content, filename):
    """
    Rename _engine to _index_engine, avoiding false positives.
    """
    lines = content.split('\n')
    modified_lines = []
    
    # Patterns to avoid
    avoid_patterns = [
        r'_engine_type',
        r'_engine_types',
        r'_engine_target',
        r'_engine_specified',
        r'_make_engine',
        r'_set_with_engine',
        r'_get_engine_target',
        r'_engines\b',  # Multiple engines
    ]
    
    for line in lines:
        # Skip if this line has patterns we should avoid
        skip_line = False
        for pattern in avoid_patterns:
            if re.search(pattern, line):
                skip_line = True
                break
        
        if skip_line:
            modified_lines.append(line)
        else:
            # Replace _engine with _index_engine but not in string comments about engines
            # Replace: @property def _engine, def _engine, ._engine, self._engine
            modified_line = re.sub(r'\b_engine\b', '_index_engine', line)
            modified_lines.append(modified_line)
    
    return '\n'.join(modified_lines)

# Process the index-related files

    ('pandas_base.py', 'pandas/core/indexes/base.py'),

    ('pandas_category.py', 'pandas/core/indexes/category.py'),
]
