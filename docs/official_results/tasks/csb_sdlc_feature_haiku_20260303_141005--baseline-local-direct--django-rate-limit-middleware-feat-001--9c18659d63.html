from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that provides configurable per-IP request rate limiting
    using Django's cache framework.
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.requests_limit = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.window_size = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

    def get_client_ip(self, request):
        """
        Get the client's IP address from the request.
        Checks X-Forwarded-For header for proxied requests.
        """
        x_forwarded_for = request.META.get("HTTP_X_FORWARDED_FOR")
        if x_forwarded_for:
            return x_forwarded_for.split(",")[0].strip()
        return request.META.get("REMOTE_ADDR")

    def process_request(self, request):
        """
        Check if the client has exceeded the rate limit.
        Returns a 429 response if the limit is exceeded.
        """
        client_ip = self.get_client_ip(request)
        cache_key = f"ratelimit:{client_ip}"

        # Get current request count
        request_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if request_count >= self.requests_limit:
            return HttpResponseTooManyRequests()

        # Increment counter with expiry equal to window size
        cache.set(cache_key, request_count + 1, self.window_size)

        return None
