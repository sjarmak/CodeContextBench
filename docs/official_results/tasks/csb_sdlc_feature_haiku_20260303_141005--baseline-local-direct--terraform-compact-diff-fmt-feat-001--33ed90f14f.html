// CompactDiffFormatter produces a condensed, summary-style diff output
// for terraform plan, showing only resource-level changes without
// attribute-level detail.
type CompactDiffFormatter struct{}

// Format returns a compact formatted summary of the plan's resource changes.
// It groups changes by action type and outputs one line per resource change,
// ending with a summary of the total changes.
func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *terraform.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	var buf bytes.Buffer

	// Group resources by action
	actionGroups := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)
	for _, rc := range plan.Changes.Resources {
		if rc.DeposedKey != "" {
			// Skip deposed resources
			continue
		}
		actionGroups[rc.Action] = append(actionGroups[rc.Action], rc)
	}

	// Define the order in which to display actions
	actionOrder := []plans.Action{
		plans.Create,
		plans.Update,
		plans.Delete,
		plans.DeleteThenCreate,
		plans.CreateThenDelete,
		plans.CreateThenForget,
		plans.Forget,
		plans.Read,
		plans.NoOp,
	}
