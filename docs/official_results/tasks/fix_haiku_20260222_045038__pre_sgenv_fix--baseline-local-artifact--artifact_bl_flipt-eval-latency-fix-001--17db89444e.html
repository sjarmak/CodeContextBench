// DurationTracker records evaluation durations and provides statistics.
// It is safe for concurrent use.
type DurationTracker struct {
	mu        sync.Mutex
	durations []time.Duration
}

// Record records a duration for an evaluation.
func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Count returns the total number of recorded evaluations.
func (dt *DurationTracker) Count() int {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	return len(dt.durations)
}

// Average returns the average duration across all evaluations.
// Returns 0 if no evaluations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.Lock()
	defer dt.mu.Unlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}

	return total / time.Duration(len(dt.durations))
}

// P99 returns the 99th percentile duration across all evaluations.
// Returns 0 if no evaluations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.Lock()
	defer dt.mu.Unlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Create a sorted copy to avoid modifying original data
	sorted := make([]time.Duration, len(dt.durations))
	copy(sorted, dt.durations)
	sort.Slice(sorted, func(i, j int) bool {
		return sorted[i] < sorted[j]
	})

	// Calculate 99th percentile index
	// Using ceil(0.99 * n) formula for the percentile position
	index := (len(sorted) * 99) / 100
	if index >= len(sorted) {
		index = len(sorted) - 1
	}

	return sorted[index]
}
