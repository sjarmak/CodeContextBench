import "sync/atomic"

// MetricsTracker records evaluation throughput counters in a thread-safe manner.
type MetricsTracker struct {
	total  atomic.Int64
	match  atomic.Int64
	errors atomic.Int64
}

// RecordEvaluation increments the total evaluations counter.
func (m *MetricsTracker) RecordEvaluation() {
	m.total.Add(1)
}

// RecordMatch increments the match counter.
func (m *MetricsTracker) RecordMatch() {
	m.match.Add(1)
}

// RecordError increments the error counter.
func (m *MetricsTracker) RecordError() {
	m.errors.Add(1)
}

// Total returns the number of evaluations processed.

	return m.total.Load()
