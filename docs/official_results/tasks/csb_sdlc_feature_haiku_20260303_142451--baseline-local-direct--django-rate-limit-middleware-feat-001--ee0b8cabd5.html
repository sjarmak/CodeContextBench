from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that implements per-IP request rate limiting using Django's cache framework.

    Configuration options (in settings.py):
    - RATE_LIMIT_REQUESTS: Maximum number of requests allowed per window (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.requests_limit = getattr(settings, 'RATE_LIMIT_REQUESTS', 100)
        self.window_size = getattr(settings, 'RATE_LIMIT_WINDOW', 3600)

    def process_request(self, request):
        """
        Check if the client IP has exceeded the rate limit.
        Returns HttpResponseTooManyRequests (429) if limit exceeded, None otherwise.
        """
        client_ip = self._get_client_ip(request)
        cache_key = f"rate_limit:{client_ip}"

        try:
            request_count = cache.incr(cache_key)
        except ValueError:
            # Key doesn't exist, set it to 1 with the window size as timeout
            cache.set(cache_key, 1, self.window_size)
            request_count = 1

        if request_count > self.requests_limit:
            return HttpResponseTooManyRequests()

        return None

    def _get_client_ip(self, request):
        """
        Extract the client IP address from the request.
        """
        return request.META.get('REMOTE_ADDR', '0.0.0.0')

from django.core.cache import cache


def get_response(request):

class RateLimitMiddlewareTest(SimpleTestCase):
    """Test cases for RateLimitMiddleware."""

    def setUp(self):
        """Clear cache before each test."""
        cache.clear()
        self.rf = RequestFactory()

    def tearDown(self):
        """Clear cache after each test."""
        cache.clear()

    @override_settings(RATE_LIMIT_REQUESTS=5, RATE_LIMIT_WINDOW=3600)
    def test_requests_within_limit(self):
        """Test that requests within the limit are allowed."""
        middleware = RateLimitMiddleware(get_response)

        for i in range(5):
            request = self.rf.get('/', REMOTE_ADDR='127.0.0.1')
            response = middleware.process_request(request)
            # Requests within limit should return None (allow processing)
            self.assertIsNone(response, f"Request {i+1} should be allowed")

    @override_settings(RATE_LIMIT_REQUESTS=5, RATE_LIMIT_WINDOW=3600)
    def test_requests_exceeding_limit(self):
        """Test that requests exceeding the limit are blocked."""
        middleware = RateLimitMiddleware(get_response)

        # Make requests up to the limit
        for i in range(5):
            request = self.rf.get('/', REMOTE_ADDR='127.0.0.1')
            response = middleware.process_request(request)
            self.assertIsNone(response)

        # Next request should be blocked
        request = self.rf.get('/', REMOTE_ADDR='127.0.0.1')
