package evaluation

import (
	"sync"
)

// Tracker is an in-process metrics tracker for the evaluation engine.
// It records evaluation outcomes including total evaluations, matches, and errors.
// It is thread-safe for concurrent access.
type Tracker struct {
	mu         sync.RWMutex
	total      int64 // total evaluations processed
	matchCount int64 // evaluations that resulted in a successful match
	errorCount int64 // evaluations that failed with errors
}

// NewTracker creates a new evaluation metrics tracker.
func NewTracker() *Tracker {
	return &Tracker{}
}

// RecordEvaluation records a completed evaluation outcome.
// If err is not nil, it records an error.
// If match is true, it records a match.
// It always increments the total evaluation count.
func (t *Tracker) RecordEvaluation(match bool, err error) {
	t.mu.Lock()
	defer t.mu.Unlock()

	t.total++
	if err != nil {
		t.errorCount++
	} else if match {
		t.matchCount++
	}
}

// Metrics returns a snapshot of the current metric values.
type Metrics struct {
	Total      int64
	MatchCount int64
	ErrorCount int64
}

// GetMetrics returns the current metric values in a thread-safe manner.
func (t *Tracker) GetMetrics() Metrics {
	t.mu.RLock()
	defer t.mu.RUnlock()

	return Metrics{
		Total:      t.total,
		MatchCount: t.matchCount,
		ErrorCount: t.errorCount,
	}
}
