type PolicyAuditLogger struct {
	mutex lock.Mutex
	log   *logrus.Logger

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified log level.
// If output is nil, the logger will use the standard Cilium logging framework.
func NewPolicyAuditLogger(level logrus.Level, output io.Writer) *PolicyAuditLogger {
	logger := logrus.New()
	logger.SetLevel(level)

	if output != nil {
		logger.SetOutput(output)
		logger.SetFormatter(&logrus.JSONFormatter{})
	} else {
		// Use default Cilium logger when no output is specified
		logger = logrus.StandardLogger()
		logger.SetLevel(level)
	}

	return &PolicyAuditLogger{
		log: logger,
	}
}
