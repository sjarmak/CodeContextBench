// SPDX-License-Identifier: Apache-2.0

package policy

import (
	"context"
	"io"

	"github.com/sirupsen/logrus"

	"github.com/cilium/cilium/pkg/identity"
	"github.com/cilium/cilium/pkg/lock"
	"github.com/cilium/cilium/pkg/logging/logfields"
)

// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes.
type PolicyAuditLogger struct {
	mutex lock.Mutex
	log   *logrus.Logger
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified log level.
// If output is nil, the logger will use the standard Cilium logging framework.
func NewPolicyAuditLogger(level logrus.Level, output io.Writer) *PolicyAuditLogger {
	logger := logrus.New()
	logger.SetLevel(level)

	if output != nil {
		logger.SetOutput(output)
		logger.SetFormatter(&logrus.JSONFormatter{})
	} else {
		// Use default Cilium logger when no output is specified
		logger = logrus.StandardLogger()
		logger.SetLevel(level)
	}

	return &PolicyAuditLogger{
		log: logger,
	}
}

// LogDecision logs a policy decision with the verdict and reason.
// This method is thread-safe.
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict string, reason string) {
